<script lang="ts">
  export let size = "15";
  export let variation: "solid" | "outline" = "outline";
  let svgpath: string;
  let svgoutline = `replace_svg_outline`;
  let svgsolid = `replace_svg_solid`;
  switch (variation) {
    case "outline":
      svgpath = svgoutline;
      break;
    case "solid":
      svgpath = svgsolid;
      break;
    default:
      svgpath = svgoutline;
  }
</script>

<svg
  xmlns="http://www.w3.org/2000/svg"
  width={size}
  height={size}
  fill="none"
  class={$$props.class}
  {...$$restProps}
  aria-label={ariaLabel}
  viewBox="0 0 15 15"
>
  {@html svgpath}
</svg>
